void handleMenuChoice(Database *db, int choice) {
    char searchStr[100];
    int index, minTracks, maxDuration;
    MusicAlbum album;
    
    switch(choice) {
        case 1:
            displayAllAlbums(db);
            break;
            
        case 2:
            album = inputAlbum();
            if (addAlbum(db, &album)) {
                printf("–ê–ª—å–±–æ–º –¥–æ–±–∞–≤–ª–µ–Ω —É—Å–ø–µ—à–Ω–æ!\n");
            }
            break;
            
        case 3:
            printf("–í–≤–µ–¥–∏—Ç–µ –Ω–æ–º–µ—Ä –∞–ª—å–±–æ–º–∞ –¥–ª—è —É–¥–∞–ª–µ–Ω–∏—è: ");
            scanf("%d", &index);
            getchar();
            if (deleteAlbum(db, index - 1)) {
                printf("–ê–ª—å–±–æ–º —É–¥–∞–ª–µ–Ω —É—Å–ø–µ—à–Ω–æ!\n");
            }
            break;
            
        case 4:
            printf("–í–≤–µ–¥–∏—Ç–µ –Ω–æ–º–µ—Ä –∞–ª—å–±–æ–º–∞ –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è: ");
            scanf("%d", &index);
            getchar();
            album = inputAlbum();
            if (updateAlbum(db, index - 1, &album)) {
                printf("–ê–ª—å–±–æ–º –æ–±–Ω–æ–≤–ª–µ–Ω —É—Å–ø–µ—à–Ω–æ!\n");
            }
            break;
            
        case 5:
            printf("–í–≤–µ–¥–∏—Ç–µ –Ω–∞–∑–≤–∞–Ω–∏–µ –¥–ª—è –ø–æ–∏—Å–∫–∞: ");
            fgets(searchStr, sizeof(searchStr), stdin);
            searchStr[strcspn(searchStr, "\n")] = 0;
            searchByTitle(db, searchStr);
            break;
            
        case 6:
            printf("–í–≤–µ–¥–∏—Ç–µ —Å—Ç–∏–ª—å –¥–ª—è –ø–æ–∏—Å–∫–∞: ");
            fgets(searchStr, sizeof(searchStr), stdin);
searchStr[strcspn(searchStr, "\n")] = 0;
            searchByStyle(db, searchStr);
            break;
            
        case 7:
            printf("–ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç—Ä–µ–∫–æ–≤: ");
            scanf("%d", &minTracks);
            printf("–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –ø—Ä–æ–¥–æ–ª–∂–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å (–º–∏–Ω—É—Ç): ");
            scanf("%d", &maxDuration);
            getchar();
            filterByTracksAndDuration(db, minTracks, maxDuration);
            break;
            
        case 8:
            sortByYear(db);
            break;
            
        case 9:
            saveDatabase(db, FILENAME);
            break;
            
        case 10:
            loadDatabase(db, FILENAME);
            break;
            
        case 11:
            exportToText(db, "albums_export.txt");
            break;
            
        case 0:

    }
